116 int momenta::set(){
117 
118  int i,j;
119  int ig, jg;
120  double sargx, sargy;
121 
122  for(i = 0; i < Nxl; i++){
123  for(j = 0; j < Nyl; j++){
124 
125  ig = (i+pos_x*Nxl);
126  jg = (j+pos_y*Nyl);
127 
128 
129  //constructs the different kinds of momenta that are needed:
130  //phat2 = sum_\mu 4 sin(pi k_\mu / N_\mu)^2
131  //pbar2 = sum_\mu sin(2pi k_\mu / N_\mu)^2
132  //pbar_\mu = sin(2pi k_\mu / N_\mu)
133 
134  sargx = sin( M_PI * ig / (1.0 * Nx) );
135  sargy = sin( M_PI * jg / (1.0 * Ny) );
136 
137  tbl_phat2[i*Nyl+j] = 4.0*pow(sargx,2.0) + 4.0*pow(sargy,2.0);
138  tbl_phatx[i*Nyl+j] = sargx;
139  tbl_phaty[i*Nyl+j] = sargy;
140 
141  sargx = sin( 2.0 * M_PI * ig / (1.0 * Nx) );
142  sargy = sin( 2.0 * M_PI * jg / (1.0 * Ny) );
143 
144  tbl_pbar2[i*Nyl+j] = pow(sargx,2.0) + pow(sargy,2.0);
145  tbl_pbarx[i*Nyl+j] = sargx;
146  tbl_pbary[i*Nyl+j] = sargy;
147  }
148  }
149 
150 return 1;
151 }
